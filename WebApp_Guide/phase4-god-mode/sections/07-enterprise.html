Access Control

Role-Based Access Control (RBAC)

# .gemini/enterprise/rbac.yaml
roles:
  developer:
    agents:
      - code-reviewer: [read, execute]
      - docs-writer: [read, execute]
    mcp:
      - filesystem: [read]
      - git: [read, write]

  tech-lead:
    extends: developer
    agents:
      - security-scanner: [read, execute]
      - deployment-bot: [read, execute]
    mcp:
      - postgres: [read]

  admin:
    extends: tech-lead
    agents:
      - "*": [read, write, execute, delete]
    mcp:
      - "*": [read, write]
    settings:
      - can_modify_rbac: true
      - can_view_audit_logs: true

Audit Logging

# .gemini/enterprise/audit.yaml
audit:
  log_level: detailed
  retention: 365d

  events:
    - agent_execution
    - mcp_tool_call
    - file_modification
    - config_change
    - permission_change

  format:
    timestamp: ISO8601
    user: email
    action: string
    resource: path
    result: success|failure
    metadata: json

  export:
    - type: elasticsearch
      endpoint: https://logs.company.com
    - type: s3
      bucket: audit-logs-prod
